# WeightLossRankings.org Phase 1 MVP — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a deployable affiliate marketing site with homepage, interactive comparison tool, provider reviews, "best of" rankings, savings calculator with email capture, 5 state pages, and a blog — seeded with 10 GLP-1 provider datasets.

**Architecture:** Next.js 14 App Router with Sanity CMS as the content backend. All content pages are statically generated at build time via `generateStaticParams`. The comparison tool (`/compare`) uses client-side filtering over pre-fetched provider data. Email capture uses ConvertKit's API. Deployed on Vercel.

**Tech Stack:** Next.js 14 (App Router), Tailwind CSS v3, Sanity v3 (free tier), ConvertKit API, Vercel, TypeScript

**Spec:** `docs/superpowers/specs/2026-04-05-weightlossrankings-design.md`

- [ ] **Step 2: Create scoring utility**

Create `src/lib/scoring.ts`:

```ts
import { Scores } from "./types";

const WEIGHTS: Record<keyof Scores, number> = {
  value: 0.25,
  effectiveness: 0.25,
  ux: 0.15,
  trust: 0.15,
  accessibility: 0.1,
  support: 0.1,
};

export function computeOverallScore(scores: Scores): number {
  let total = 0;
  for (const [key, weight] of Object.entries(WEIGHTS)) {
    total += (scores[key as keyof Scores] ?? 0) * weight;
  }
  return Math.round(total * 10) / 10;
}

export function getScoreColor(score: number): string {
  if (score >= 8) return "text-brand-success";
  if (score >= 6) return "text-brand-warning";
  return "text-red-500";
}

export const SCORE_DIMENSIONS = [
  { key: "value" as const, label: "Value", weight: "25%" },
  { key: "effectiveness" as const, label: "Effectiveness", weight: "25%" },
  { key: "ux" as const, label: "User Experience", weight: "15%" },
  { key: "trust" as const, label: "Trust & Safety", weight: "15%" },
  { key: "accessibility" as const, label: "Accessibility", weight: "10%" },
```

```ts
import { Provider, FilterState, SortOption } from "./types";
import { computeOverallScore } from "./scoring";

export function filterProviders(providers: Provider[], filters: FilterState): Provider[] {
  let result = [...providers];

  if (filters.category) {
    result = result.filter((p) => p.category === filters.category);
  }
  if (filters.drug_type) {
    result = result.filter((p) =>
      p.pricing.some((pr) => pr.dose.toLowerCase().includes(filters.drug_type!.toLowerCase()))
    );
  }
  if (filters.form) {
    result = result.filter((p) => p.pricing.some((pr) => pr.form === filters.form));
  }
  if (filters.state) {
    result = result.filter((p) => p.states_available.includes(filters.state!));
  }
  if (filters.dose) {
    result = result.filter((p) => p.pricing.some((pr) => pr.dose === filters.dose));
  }
  if (filters.features && filters.features.length > 0) {
    result = result.filter((p) =>
      filters.features!.every((f) => p.features.includes(f))
    );
  }
  if (filters.max_price !== undefined) {
    result = result.filter((p) =>
      p.pricing.some((pr) => (pr.promo_price ?? pr.monthly_cost) <= filters.max_price!)
    );
  }

  return sortProviders(result, filters.sort, filters.dose);
}

function sortProviders(providers: Provider[], sort: SortOption, dose?: string): Provider[] {
  switch (sort) {
    case "cheapest":
      return providers.sort((a, b) => getMinPrice(a, dose) - getMinPrice(b, dose));
    case "highest-rated":
      return providers.sort((a, b) => computeOverallScore(b.scores) - computeOverallScore(a.scores));
    case "most-reviewed":
      return providers.sort(
        (a, b) =>
          (b.external_reviews?.trustpilot_count ?? 0) - (a.external_reviews?.trustpilot_count ?? 0)
      );
    default:
      return providers;
  }
}

function getMinPrice(provider: Provider, dose?: string): number {
  const prices = dose
    ? provider.pricing.filter((p) => p.dose === dose)
    : provider.pricing;
  if (prices.length === 0) return Infinity;
  return Math.min(...prices.map((p) => p.promo_price ?? p.monthly_cost));
}
```

```tsx
"use client";

import { useState } from "react";
import { Provider } from "@/lib/types";
import { MedicationStep } from "./MedicationStep";
import { DoseStep } from "./DoseStep";
import { CostStep } from "./CostStep";
import { EmailGate } from "./EmailGate";
import { SavingsResults } from "./SavingsResults";

export interface CalcState {
  medication: string;
  dose: string;
  currentCost: number | null;
  email: string;
}

interface StepWizardProps {
  providers: Provider[];
}

export function StepWizard({ providers }: StepWizardProps) {
  const [step, setStep] = useState(0);
  const [state, setState] = useState<CalcState>({
    medication: "",
    dose: "",
    currentCost: null,
    email: "",
  });

  const steps = [
    <MedicationStep key="med" value={state.medication} onNext={(v) => { setState({ ...state, medication: v }); setStep(1); }} />,
    <DoseStep key="dose" medication={state.medication} value={state.dose} onNext={(v) => { setState({ ...state, dose: v }); setStep(2); }} onBack={() => setStep(0)} />,
    <CostStep key="cost" value={state.currentCost} onNext={(v) => { setState({ ...state, currentCost: v }); setStep(3); }} onBack={() => setStep(1)} />,
    <EmailGate key="email" onNext={(email) => { setState({ ...state, email }); setStep(4); }} onBack={() => setStep(2)} />,
    <SavingsResults key="results" state={state} providers={providers} />,
  ];

  return (
    <div>
      {/* Progress bar */}
      <div className="flex gap-1.5 mb-8">
        {[0, 1, 2, 3, 4].map((i) => (
          <div key={i} className={`h-1.5 flex-1 rounded-full transition-all ${i <= step ? "bg-brand-gradient" : "bg-brand-violet/10"}`} />
        ))}
      </div>
      {steps[step]}
    </div>
  );
}
```

```tsx
"use client";

import { useState } from "react";

interface Props {
  onNext: (email: string) => void;
  onBack: () => void;
}

export function EmailGate({ onNext, onBack }: Props) {
  const [email, setEmail] = useState("");
  const [loading, setLoading] = useState(false);

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault();
    setLoading(true);
    await fetch("/api/subscribe", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ email }),
    });
    setLoading(false);
    onNext(email);
  }

  return (
    <div className="text-center max-w-md mx-auto">
      <div className="text-5xl mb-4">💰</div>
      <h2 className="font-heading text-2xl font-bold text-brand-text-primary">Your savings report is ready!</h2>
      <p className="text-brand-text-secondary mt-2">Enter your email to see your personalized savings breakdown and provider recommendations.</p>
      <form onSubmit={handleSubmit} className="mt-6 space-y-3">
        <input
          type="email"
          required
          value={email}
          onChange={(e) => setEmail(e.target.value)}
          placeholder="you@example.com"
          className="w-full rounded-xl border border-brand-violet/15 px-4 py-4 text-base outline-none focus:ring-2 focus:ring-brand-violet/30"
        />
        <button
          type="submit"
          disabled={loading}
          className="w-full rounded-full bg-brand-gradient px-8 py-4 text-white font-semibold shadow-lg tap-target disabled:opacity-50"
        >
          {loading ? "Loading..." : "See My Savings"}

      </form>
      <button onClick={onBack} className="mt-4 text-sm text-brand-text-secondary hover:text-brand-violet tap-target">
```

Create `src/components/calculator/SavingsResults.tsx`:

```tsx
import { CalcState } from "./StepWizard";
import { Provider } from "@/lib/types";
import { ProviderCard } from "@/components/providers/ProviderCard";

const BRAND_PRICES: Record<string, number> = {
  wegovy: 1349,
  ozempic: 935,
  mounjaro: 1023,
  zepbound: 1059,
};

interface Props {
  state: CalcState;
  providers: Provider[];
}

export function SavingsResults({ state, providers }: Props) {
  const currentMonthly = state.currentCost ?? BRAND_PRICES[state.medication] ?? 1000;

  const providerSavings = providers
    .map((p) => {
      const matching = p.pricing.filter((pr) => pr.form === "compounded");
      if (matching.length === 0) return null;
      const cheapest = Math.min(...matching.map((pr) => pr.promo_price ?? pr.monthly_cost));
      return { provider: p, cheapest, savings: currentMonthly - cheapest };
    })
    .filter((x): x is NonNullable<typeof x> => x !== null && x.savings > 0)
    .sort((a, b) => b.savings - a.savings);

  const topSavings = providerSavings[0]?.savings ?? 0;

  return (
    <div>
      <div className="text-center mb-8">
        <h2 className="font-heading text-2xl font-bold text-brand-text-primary">Your Savings Report</h2>
        {topSavings > 0 ? (
          <div className="mt-4">
            <p className="text-brand-text-secondary">You could save up to</p>
            <p className="text-5xl font-bold text-gradient mt-2">${topSavings.toLocaleString()}/mo</p>
            <p className="text-brand-text-secondary mt-1">
              That's <span className="font-bold text-brand-success">${(topSavings * 12).toLocaleString()}/year</span> by switching to a compounded alternative
            </p>
          </div>
        ) : (
          <p className="text-brand-text-secondary mt-4">Here are the best-priced providers for your medication.</p>
        )}
      </div>

      <div className="space-y-4">
        {(providerSavings.length > 0 ? providerSavings.map((ps) => ps.provider) : providers.slice(0, 5)).map(
          (provider) => (
            <ProviderCard key={provider._id} provider={provider} selectedDose={state.dose} />
          )
        )}
      </div>
    </div>
  );
}
```
